cmake_minimum_required(VERSION 3.10)

# Set the project name
project(PropertyTree)

# Specify the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add googletest as a subdirectory
add_subdirectory(deps/googletest-main)

# Add flatbuffers as a subdirectory
add_subdirectory(deps/flatbuffers-master)

add_subdirectory(src)

include_directories(
        ${PROJECT_SOURCE_DIR}/deps/googletest-main/googletest/include
        ${PROJECT_SOURCE_DIR}/deps/flatbuffers-master/include
        ${PROJECT_SOURCE_DIR}/src
        ${PROJECT_SOURCE_DIR}/
)

enable_testing()
# Add executable
add_executable(PropertyTree main.cpp)

# Link the executable to the FlatBuffers library and googletest
target_link_libraries(PropertyTree PRIVATE tcp_service flatbuffers gtest gtest_main)

# Add test executable
add_executable(
    test_property_tree
    tests/test_property_tree.cpp
        tests/utils.cpp)

# Link the test executable to googletest
target_link_libraries(test_property_tree PRIVATE gtest gtest_main tcp_service flatbuffers)


# Add test
add_test(NAME test_property_tree COMMAND test_property_tree)
